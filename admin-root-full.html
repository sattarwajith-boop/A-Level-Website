Firebase Security Rules

Copy and paste these rules into your Firebase console. Admin writes are allowed only after Firebase Authentication login.

Firestore Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isSignedIn() {
      return request.auth != null;
    }

    function isAdmin() {
      return isSignedIn();
    }

    match /resources/{resourceId} {
      allow read: if resource.data.status == "published" || isAdmin();
      allow create, update, delete: if isAdmin();
    }

    match /announcements/{announcementId} {
      allow read: if resource.data.status == "active" || isAdmin();
      allow create, update, delete: if isAdmin();
    }

    match /subjects/{subjectId} {
      allow read: if true;
      allow create, update, delete: if isAdmin();
    }

    match /siteSettings/{docId} {
      allow read: if true;
      allow create, update, delete: if isAdmin();
    }

    match /brokenReports/{reportId} {
      allow create: if true;
      allow read, update, delete: if isAdmin();
    }
  }
}

Storage Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {

    function isSignedIn() {
      return request.auth != null;
    }

    function isAdmin() {
      return isSignedIn();
    }

    match /resources/{allPaths=**} {
      allow read: if true;
      allow write: if isAdmin()
        && request.resource.size < 100 * 1024 * 1024;
    }
  }
}
Setup Steps
  1. Open Firebase Console
  2. Go to Firestore -> Rules -> paste Firestore rules above
  3. Go to Storage -> Rules -> paste Storage rules above
  4. Publish both rule sets
  5. Create your admin user in Firebase Authentication
  6. Keep public sign-up pages disabled so only manually created Firebase users can enter admin
